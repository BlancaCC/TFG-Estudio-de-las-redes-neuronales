% Teorema 2.2 
\begin{teorema}
    Para cualquier función continua no constate $G$, $r \in \N$ y
    medida de probabilidad $\mu$ o $(\R^r, B^r)$, 
    se tiene que $\rrnng$ es uniformemente denso en compactos
    en $\fC$ y denso en $\fM$ de acorde a la distancia $\dist$. 
\end{teorema}
